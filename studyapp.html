class GoalType(Enum):
    PROMOTION     = "promotion"   # 現職での昇進・昇給
    CAREER_CHANGE = "change"      # 業種・職種の転換
    INDEPENDENT   = "independent" # 独立・フリーランス化
    STARTUP       = "startup"     # 起業・スタートアップ
    OVERSEAS      = "overseas"    # 海外就労・移住
    STABILITY     = "stability"   # 安定・セカンドキャリア


@dataclass
class UserProfile:
    """ユーザーの入力情報"""

@dataclass
class Certification:
    """資格マスターデータ"""
    id:                   str
    name:                 str
    org:                  str
    category:             str
    tags:                 List[str]
    study_hours_required: int         # 合格に必要な総学習時間
    cost:                 int         # 受験費用（円）
    avg_salary_increase:  float       # 平均年収増加率
    demand_score:         float       # 市場需要スコア 0-100
    difficulty:           int         # 難易度 1-5
    goal_match:           List[str]    # 対応ゴールタイプ
    interest_match:       List[str]    # 対応興味分野


@dataclass
class OpportunityCost:

    payback_months:       float       # 投資回収期間（月）
    roi_score:            float       # ROIスコア 0-100
    time_to_complete:     float       # 完了までの月数

# -------------------------------------------------
# 3. 機会費用・ROI 計算エンジン
#
# 計算式:
#   時給 = 年収(円) / 2080
#   機会費用 = 時給 x 必要学習時間
#   総投資額 = 機会費用 + 受験費 + 教材費見積
#   期待年収増加 = 年収 x 平均昇給率 x ゴール係数
#   ROIスコア = clip((期待増加-総投資)/総投資 x 20+50, 0, 100)
# -------------------------------------------------

ANNUAL_WORK_HOURS = 2080   # 年間標準労働時間（週40h x 52週）

GOAL_MULTIPLIERS = {
    "promotion":   1.0,
    "change":      1.2,
    "independent": 1.5,
    "startup":     2.0,
    "overseas":    1.3,
    "stability":   0.8,
}


def calculate_oc(profile: UserProfile, cert: Certification) -> OpportunityCost:
    """機会費用と ROI を計算する"""
    MAN_YEN = 10_000
    hourly_rate   = profile.annual_income * MAN_YEN / ANNUAL_WORK_HOURS
    weeks_needed  = cert.study_hours_required / profile.study_hours_per_week
    months_needed = weeks_needed / 4.3
    study_cost    = cert.study_hours_required * hourly_rate
    material_cost = profile.monthly_budget * MAN_YEN * months_needed * 0.3
    total_inv     = study_cost + cert.cost + material_cost
    multiplier    = GOAL_MULTIPLIERS.get(profile.goal_type.value, 1.0)
    expected_gain = profile.annual_income * MAN_YEN * cert.avg_salary_increase * multiplier
    monthly_gain  = expected_gain / 12
    payback       = total_inv / monthly_gain if monthly_gain > 0 else float("inf")
    raw_roi       = (expected_gain - total_inv) / (total_inv + 1)
    roi_score     = min(100.0, max(0.0, raw_roi * 20 + 50))
    return OpportunityCost(
        hourly_rate          = round(hourly_rate,    0),
        total_study_cost     = round(study_cost,     0),
        exam_cost            = cert.cost,
        total_investment     = round(total_inv,      0),
        expected_annual_gain = round(expected_gain,  0),
        payback_months       = round(payback,        1),
        roi_score            = round(roi_score,      1),
        time_to_complete     = round(months_needed,  1),
    )


# -------------------------------------------------
# 4. スコアリング・ランキング
#
#   総合スコア（100点満点）の内訳:
#     ROIスコア      40%
#     市場需要スコア  20%
#     ゴール適合度   20%
#     興味適合度     15%
#     実現可能性      5%
# -------------------------------------------------

def _feasibility_score(profile: UserProfile, cert: Certification, months: float) -> float:
    period_ok = profile.study_months == 0 or months <= profile.study_months
    budget_ok = cert.cost <= profile.monthly_budget * 10_000 * months
    if period_ok and budget_ok: return 100.0
    if period_ok or  budget_ok: return  50.0
    return 0.0


def _total_score(profile: UserProfile, cert: Certification, oc: OpportunityCost) -> float:
    goal_match = 100.0 if profile.goal_type.value in cert.goal_match else 0.0
    if profile.interests:
        matched = len(set(profile.interests) & set(cert.interest_match))
        interest_match = matched / len(profile.interests) * 100
    else:
        interest_match = 50.0
    feasibility = _feasibility_score(profile, cert, oc.time_to_complete)
    return (
        oc.roi_score        * 0.40
        + cert.demand_score * 0.20
        + goal_match        * 0.20
        + interest_match    * 0.15
        + feasibility       * 0.05
    )


def rank_certifications(
    profile: UserProfile,
    db: List[Certification] = CERTIFICATIONS,
    top_n: int = 6,
) -> List[tuple]:
    """全資格をスコアリングして降順ソートし上位 top_n を返す"""
    scored = []
    for cert in db:
        oc    = calculate_oc(profile, cert)
        score = _total_score(profile, cert, oc)
        scored.append((cert, oc, score))
    scored.sort(key=lambda x: x[2], reverse=True)
    return scored[:top_n]


def analyse_factors(profile: UserProfile) -> dict:
    """ユーザーの ROI 影響ファクターを 0-100 でスコア化する"""
    income_gap = profile.target_income - profile.annual_income
    pot = (income_gap / profile.annual_income * 100) if profile.annual_income > 0 else 0
    return {
        "時間効率":        min(100, profile.study_hours_per_week * 2.5),
        "投資余力":        min(100, profile.monthly_budget * 10),
        "年収ポテンシャル": min(100, max(0, pot)),
        "ゴール明確度":    85.0 if profile.goal_type else 40.0,
        "学習継続性":      min(100, math.log1p(profile.study_hours_per_week) * 30),
    }

def _input_int(prompt: str, lo: int, hi: int) -> int:
    while True:
        try:
            v = int(input(prompt))
            if lo <= v <= hi: return v
            print(f"  {lo}～{hi} の値を入力してください。")
        except ValueError:
            print("  数値を入力してください。")


def _input_float(prompt: str, lo: float, hi: float) -> float:
    while True:
        try:
            v = float(input(prompt))
            if lo <= v <= hi: return v
            print(f"  {lo}～{hi} の値を入力してください。")
        except ValueError:
            print("  数値を入力してください。")


def collect_profile() -> UserProfile:
    print("\n" + "="*52)
    print("  SkillRoi - 学習投資対効果診断")
    print("="*52)
    name   = input("\nお名前: ").strip() or "ユーザー"
    age    = _input_int("年齢: ", 18, 70)
    income = _input_int("現在の年収（万円）: ", 0, 5000)
    hours  = _input_float("週の学習可能時間（h）: ", 0.5, 40)
    budget = _input_int("月の学習予算（万円, 0=なし）: ", 0, 100)
    target = _input_int("目標年収（万円）: ", income, 5000)
    period = _input_int("希望学習期間（月, 0=制限なし）: ", 0, 60)

    print("\n目標タイプを選んでください:")
    for n, _, label in GOAL_MENU:
        print(f"  {n}. {label}")
    g_key = _input_int("番号: ", 1, len(GOAL_MENU))
    goal  = GOAL_MENU[g_key - 1][1]

    print("\n興味のある分野を選んでください（複数可、カンマ区切り 例: 1,3,7）:")
    for n, _, label in INTEREST_MENU:
        print(f"  {n}. {label}")
    raw = input("番号（Enterでスキップ）: ").strip()
    interests: List[str] = []
    if raw:
        idx_map = {n: v for n, v, _ in INTEREST_MENU}
        for token in raw.split(","):
            v = idx_map.get(token.strip())
            if v: interests.append(v)

    return UserProfile(
        name=name, age=age, annual_income=income,
        study_hours_per_week=hours, goal_type=goal,
        target_income=target, interests=interests,
        monthly_budget=budget, study_months=period,
    )


def print_results(profile: UserProfile) -> None:
    ranked  = rank_certifications(profile)
    factors = analyse_factors(profile)
    hourly  = profile.annual_income * 10_000 / ANNUAL_WORK_HOURS

    print(f"\n{'='*52}")
    print(f"  診断結果 - {profile.name}さん")
    print(f"{'='*52}")
    print(f"  時給換算（機会費用基準）: ¥{hourly:,.0f} / 時間")
    print(f"\n  ROIファクター分析:")
    for k, v in factors.items():
        bar = "#" * int(v / 5) + "-" * (20 - int(v / 5))
        print(f"  {k:<12} [{bar}] {v:.0f}/100")
    print(f"\n  おすすめ資格ランキング（上位{len(ranked)}件）")
    print(f"  {'-'*48}")
    for rank, (cert, oc, score) in enumerate(ranked, 1):
        stars = "*" * cert.difficulty + "-" * (5 - cert.difficulty)
        print(f"\n  #{rank}  {cert.name}")
        print(f"       主催     : {cert.org}")
        print(f"       難易度   : [{stars}]")
        print(f"       ROIスコア: {oc.roi_score:.1f} / 100")
        print(f"       総投資額  : ¥{oc.total_investment:>12,.0f}")
        print(f"       期待年収増: ¥{oc.expected_annual_gain:>12,.0f} / 年")
        print(f"       投資回収  :  {oc.payback_months:.1f} ヶ月")
        print(f"       学習期間  :  {oc.time_to_complete:.1f} ヶ月")
    print(f"\n{'='*52}\n")
